test.describe('LLM Summaries', () => {
  test('LLM-01: collapsible summary under paragraph heading', async ({ page }) => {
    await page.goto('./gemeindeordnungen/burgenland.html');
    // Find a details element with "Vereinfachte Zusammenfassung"
    const summary = page.locator('details summary:has-text("Vereinfachte Zusammenfassung")').first();
    await expect(summary).toBeVisible();
    // Verify collapsed by default
    const details = summary.locator('..');
    await expect(details).not.toHaveAttribute('open', '');
    // Click to expand
    await summary.click();
    await expect(details).toHaveAttribute('open', '');
    // Verify summary text is visible
    const summaryText = details.locator('p');
    await expect(summaryText).toBeVisible();
    await page.screenshot({ path: 'e2e/screenshots/llm-summary-expanded.png' });
  });

  test('LLM-02: disclaimer info-box on law page', async ({ page }) => {
    await page.goto('./gemeindeordnungen/burgenland.html');
    const disclaimer = page.locator('text=keine Rechtsberatung');
    await expect(disclaimer).toBeVisible();
    await page.screenshot({ path: 'e2e/screenshots/llm-disclaimer.png' });
  });
});

test.describe('Topic Filter', () => {
  test('LLM-07: topic chips filter paragraphs', async ({ page }) => {
    await page.goto('./gemeindeordnungen/burgenland.html');
    const filterContainer = page.locator('#topic-filter');
    await expect(filterContainer).toBeVisible();

    // "Alle" chip is active by default
    const alleChip = filterContainer.locator('[data-topic="alle"]');
    await expect(alleChip).toHaveClass(/topic-chip-active/);

    // Count initial visible articles
    const allArticles = page.locator('article[data-topics]');
    const initialCount = await allArticles.count();
    expect(initialCount).toBeGreaterThan(0);

    // Click a specific topic chip
    const topicChip = filterContainer.locator('[data-topic]:not([data-topic="alle"])').first();
    const topicName = await topicChip.getAttribute('data-topic');
    await topicChip.click();

    // Verify chip becomes active
    await expect(topicChip).toHaveClass(/topic-chip-active/);
    await expect(alleChip).toHaveClass(/topic-chip-inactive/);

    // Some articles should be hidden
    const visibleArticles = page.locator('article[data-topics]:visible');
    const filteredCount = await visibleArticles.count();
    expect(filteredCount).toBeLessThanOrEqual(initialCount);

    await page.screenshot({ path: 'e2e/screenshots/topic-filter-active.png' });

    // Click "Alle" to reset
    await alleChip.click();
    const resetCount = await page.locator('article[data-topics]:visible').count();
    expect(resetCount).toBe(initialCount);

    await page.screenshot({ path: 'e2e/screenshots/topic-filter-reset.png' });
  });
});
